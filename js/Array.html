});

const eArr = arr[Symbol.iterator]();
for (let letter of eArr) {
  console.log(letter);
}

# 创建指定长度的数组

如果我们要创建长度为2的空数组 常用一下方式

const a = [];
a.length = 2;
console.log(a); // [empty × 2]

Array(2); // [empty × 2]

Array.from({length: 2}); // [undefined, undefined]

[...Array(2)]; // [undefined, undefined]

Array(2).fill(undefined); // [undefined, undefined]

Array.from({length: 5}, (v, i) => i); // [0, 1, 2, 3, 4]
[...Array(5).keys()]; // [0, 1, 2, 3, 4]

Array.from相当于两步操作 初始化+遍历

经过测试，初始数组为[0, 1, 2, 3, ...]这类数组，第二种性能会较好

# 生成二维空数组

解构赋值+map方式

[...Array(5)].map(v => []);

直接fill方式

const arr = Array(5).fill([]);

注意在这种方式下，如果修改arr[0][1] = 1，其他所有项都会同步，所以这种方式是不合适的。

# 数组去重合并

单一数组去重 可以使用Set数据结构

const set = new Set([1, 1, 2, 2]); // [1, 2]
Object.prototype.toString.call(set); // [object Set]

注意此处返回的是Set格式的数据结构，如果需要使用Array，需要使用Array.from(set)更改为Array

多个数组合并去重

function combine(...arg){ 
  let arr = [].concat.apply([], arg);  //没有去重复的新数组 
  return Array.from(new Set(arr));
} 
var m = [1, 2, 2], n = [2,3,3], l = [4, 2, 2]; 
console.log(combine(m, n, l));                     // [1, 2, 3, 4]
